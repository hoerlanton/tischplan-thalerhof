import { Injectable } from '@angular/core';
import { Http, Headers } from '@angular/http';
import 'rxjs/add/operator/map';

@Injectable()

  Http: Http;

    console.log('Task service initialized!');

    return this.http.get('imHausListe')
      .map(res => res.json());
